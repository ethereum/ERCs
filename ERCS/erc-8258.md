## Abstract

Minimal interfaces for modular smart accounts on [EIP-8141](./eip-8141.md) frame transactions.

## Specification

### Module Types

| Type | Name |
|------|------|
| 1 | Validator |
| 2 | Executor |
| 3 | Hook |
| 4 | Config |

### Structs

```solidity
struct FrameAuthContext {
    uint8 frameIndex;
    uint8 mode;
    uint8 flags;
    address target;
    bytes data;
}

struct Signature {
    uint8 scheme;
    address signer;
    bytes32 msg;
    bytes signature;
}
```

### Interfaces

```solidity
interface IFrameValidator {
    function validateFrame(
        FrameAuthContext calldata ctx,
        Signature[] calldata signatures
    ) external view returns (bool approved, uint8 scope);
}

interface IFrameExecutor {
    function execute(bytes calldata data) external returns (bytes memory);
}

interface IFrameHook {
    function preExecute(address target, uint256 value, bytes calldata data) external returns (bytes memory);
    function postExecute(bytes calldata hookData) external;
}

interface IFrameConfig {
    function readConfig(bytes32 key) external view returns (bytes memory);
    function writeConfig(bytes32 key, bytes calldata value) external;
}

interface IFrameAccount {
    function installModule(uint8 moduleType, address module, bytes calldata initData) external;
    function uninstallModule(uint8 moduleType, address module, bytes calldata deInitData) external;
    function isModuleInstalled(uint8 moduleType, address module) external view returns (bool);
}
```
